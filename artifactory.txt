pipeline {
    agent any
    tools {
        maven 'MAVEN_HOME'
    }
    
    stages{
        stage('Build'){
            steps{
                 bat 'mvn clean package'
                 archiveArtifacts artifacts: 'target/*.war', onlyIfSuccessful: true
            }
        }
        stage('Upload War To Artifactory'){
            steps{
                script{

                    def mavenPom = 'pom.xml'
                    def artifactoryRepoName = 3.0.0-SNAPSHOT ? "libs-snapshot-local" : "libs-release-local"
                    artifactoryArtifactUploader artifacts: [
                        [
                            artifactId: 'simple-app'
                            classifier: '' 
                            file: "target/simple-app-3.0.0.war" 
                            type: 'war'
                        ]
                    ], 
                    credentialsId: 'admin' 
                    groupId: 'in.javahome'
                    artifactoryUrl: 'http://localhost:8081/artifactory'
                    artifactoryVersion: '6.16.0' 
                    repository: 'artifactory-build-info' 
                    version: "3.0.0"
                    }
            }
        }
    }
}
